<script>
const NUMBERS_LEFT =[[31,22,29],[15,4,2],[11,8,10],[28,35,26],[17,6,13],[24,33,20]];
const NUMBERS_RIGHT=[[14,9,18],[32,19,21],[36,30,23],[7,12,3],[25,34,27],[5,16,1]];
const NB_PARTIES=3;

const state=[];
for(let p=0;p<NB_PARTIES;p++) state.push(makePartie());

function makePartie(){
  return {
    numero:'',
    leftCells: initCells(),
    rightCells: initCells(),
    history: [] // Historique propre à chaque tableau
  };
}

function initCells(){
  const c={};
  for(let r=0;r<6;r++) for(let col=0;col<3;col++) c[`${r}-${col}`]='white';
  return c;
}

function getRouletteColor(num) {
  if (num === 0) return 'vert';
  const noirs = [2, 4, 6, 8, 10, 11, 13, 15, 17, 20, 22, 24, 26, 28, 29, 31, 33, 35];
  return noirs.includes(num) ? 'noir' : 'rouge';
}

function renderAll(){
  const c=document.getElementById('parties-container');
  c.innerHTML='';
  for(let p=0;p<NB_PARTIES;p++) c.appendChild(renderPartie(p));
}

function checkBlock(p,row,col,isRight){
  const cells=isRight?state[p].rightCells:state[p].leftCells;
  const br=Math.floor(row/3)*3;
  
  // Analyse de la ligne dans le bloc 3x3 de ce tableau
  let reds=0;
  for(let c=0;c<3;c++) if(cells[`${row}-${c}`]==='red') reds++;
  if(reds===2) {
    for(let c=0;c<3;c++){
      const k=`${row}-${c}`;
      if(cells[k]!=='red') cells[k]='green';
    }
  }
  
  // Analyse de la colonne dans le bloc 3x3 de ce tableau
  reds=0;
  for(let r=br;r<br+3;r++) if(cells[`${r}-${col}`]==='red') reds++;
  if(reds===2) {
    for(let r=br;r<br+3;r++){
      const k=`${r}-${col}`;
      if(cells[k]!=='red') cells[k]='green';
    }
  }
}

// Recherche et active le numéro dans les carrés d'un seul tableau (p)
function simulateNumberInGrid(p, num) {
  let found = false;
  let isRight = false;
  let row = -1;
  let col = -1;

  for (let r = 0; r < 6; r++) {
    for (let c = 0; c < 3; c++) {
      if (NUMBERS_LEFT[r][c] === num) {
        found = true;
        isRight = false;
        row = r;
        col = c;
        break;
      }
    }
    if (found) break;
  }

  if (!found) {
    for (let r = 0; r < 6; r++) {
      for (let c = 0; c < 3; c++) {
        if (NUMBERS_RIGHT[r][c] === num) {
          found = true;
          isRight = true;
          row = r;
          col = c;
          break;
        }
      }
      if (found) break;
    }
  }

  if (found) {
    const cells = isRight ? state[p].rightCells : state[p].leftCells;
    const k = `${row}-${col}`;
    if (cells[k] === 'green') return;
    cells[k] = 'red';
    checkBlock(p, row, col, isRight);
  }
}

function playNumberOnTable(p, num) {
  num = parseInt(num);
  if (isNaN(num) || num < 0 || num > 36) return;
  state[p].history.push(num);
  simulateNumberInGrid(p, num);
  renderAll();
}

function undoLastNumberOnTable(p) {
  const s = state[p];
  if (s.history.length === 0) return;
  s.history.pop();
  const savedHistory = [...s.history];
  
  const oldNum = s.numero;
  state[p] = makePartie();
  state[p].numero = oldNum;
  
  for (const num of savedHistory) {
    state[p].history.push(num);
    simulateNumberInGrid(p, num);
  }
  renderAll();
}

function clearTable(p) {
  const oldNum = state[p].numero;
  state[p] = makePartie();
  state[p].numero = oldNum;
  renderAll();
}

function renderPartie(p){
  const s=state[p];
  const bloc=document.createElement('div');bloc.className='partie-bloc';

  // 1. Header (Titre du tableau + nom)
  const hdr=document.createElement('div');hdr.className='partie-header';
  const leftGrp=document.createElement('div');leftGrp.style.cssText='display:flex;align-items:center;gap:5px;';
  const lbl=document.createElement('span');lbl.className='partie-label';lbl.textContent=`T${p+1} N°:`;
  const numInp=document.createElement('input');numInp.className='partie-numero';numInp.type='text';
  numInp.value=s.numero;numInp.placeholder='Nom N°';
  numInp.addEventListener('input',e=>{state[p].numero=e.target.value;});
  leftGrp.appendChild(lbl);leftGrp.appendChild(numInp);
  hdr.appendChild(leftGrp);
  bloc.appendChild(hdr);

  // 2. Bandeau historique propre à ce tableau
  const bContainer = document.createElement('div');
  bContainer.className = 'bandeau-historique-container';
  const bLabel = document.createElement('span');
  bLabel.className = 'hist-label';
  bLabel.textContent = 'Derniers :';
  bContainer.appendChild(bLabel);
  
  const bNumeros = document.createElement('div');
  bNumeros.className = 'bandeau-numeros';
  const reversed = [...s.history].reverse();
  bNumeros.innerHTML = reversed.map(num => {
    const color = getRouletteColor(num);
    return `<span class="numero-badge ${color}">${num}</span>`;
  }).join('');
  bContainer.appendChild(bNumeros);
  bloc.appendChild(bContainer);

  // 3. Actions (Saisir 0, Annuler, RAZ)
  const actDiv = document.createElement('div');
  actDiv.className = 'table-actions';
  
  const btnZero = document.createElement('button');
  btnZero.className = 'btn-action btn-zero';
  btnZero.textContent = 'Saisir 0';
  btnZero.addEventListener('click', () => playNumberOnTable(p, 0));
  
  const btnUndo = document.createElement('button');
  btnUndo.className = 'btn-action btn-undo';
  btnUndo.textContent = 'Annuler';
  btnUndo.addEventListener('click', () => undoLastNumberOnTable(p));
  
  const btnRaz = document.createElement('button');
  btnRaz.className = 'btn-action btn-raz';
  btnRaz.textContent = 'RAZ';
  btnRaz.addEventListener('click', () => clearTable(p));
  
  actDiv.appendChild(btnZero);
  actDiv.appendChild(btnUndo);
  actDiv.appendChild(btnRaz);
  bloc.appendChild(actDiv);

  // 4. Carrés (Grilles de numéros)
  const jeuZone=document.createElement('div');jeuZone.className='jeu-zone';
  function makeGrid(isRight,rowStart){
    const cells=isRight?s.rightCells:s.leftCells;
    const nums=isRight?NUMBERS_RIGHT:NUMBERS_LEFT;
    const wrap=document.createElement('div');wrap.className='mini-grid';
    for(let r=rowStart;r<rowStart+3;r++){
      for(let c=0;c<3;c++){
        const k=`${r}-${c}`;
        const div=document.createElement('div');
        div.className='cell '+(cells[k]||'white');
        div.textContent=nums[r][c];
        div.addEventListener('click',()=>{
          const cur=cells[k];
          if(cur==='green') return;
          // Le clic sur un carré n'impacte que le tableau en cours (p)
          playNumberOnTable(p, nums[r][c]);
        });
        wrap.appendChild(div);
      }
    }
    return wrap;
  }
  const lw=document.createElement('div');lw.className='grille-wrap';
  lw.appendChild(makeGrid(false,0));lw.appendChild(makeGrid(false,3));
  const rw=document.createElement('div');rw.className='grille-wrap';
  rw.appendChild(makeGrid(true,0));rw.appendChild(makeGrid(true,3));
  jeuZone.appendChild(lw);jeuZone.appendChild(rw);

  bloc.appendChild(jeuZone);
  return bloc;
}

renderAll();
</script>
